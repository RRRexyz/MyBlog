# 车联网-基于V2X的车辆通行辅助应用(1)：绿波车速引导策略设计

基于V2X通信技术的绿波车速引导策略，实现交叉口高效通行，减少车辆延误和停车次数。通过 RSU（路侧单元）实时广播交叉口的信号灯相位与计时信息（SPAT消息）以及车道地理信息（MAP消息），OBU（车载单元）接收数据并计算车辆到达交叉口的时间。同时根据信号灯的绿波时间窗口，为车辆提供最优车速引导，提升整体通行效率。

```C++
int spat_alg(qBsmInfo *pbsmSelfInfo, qSpatInfo *pspatInfo, toCarInfoList *pstoCarInfoList)
{
    int i = 0;
    qPhaseAttr car_self_phaseInfo = {0};
    static uint32_t spat_print_time_cnt = 0;
    
    for (i=0;i<pspatInfo->PhaseNum;i++) // 遍历所有交通灯
    {
        //printf("phaseId_original %d phaseId %d \n", pbsmSelfInfo->stposOnMap.phaseId_original, pspatInfo->stPhaseInfo[i].phaseId);
        if (pbsmSelfInfo->stposOnMap.phaseId_original == pspatInfo->stPhaseInfo[i].phaseId)   
        {
            memcpy(&car_self_phaseInfo, &pspatInfo->stPhaseInfo[i], sizeof(qPhaseAttr));
            if(spat_print_time_cnt > PRINT_TIME)
            {
                my_printf("*************** [INFO] Self-Car PhaseInfo **************\n");
                my_printf("phase                   %d\n", car_self_phaseInfo.phaseId); // 交通灯编号
                my_printf("curCount                %d\n", car_self_phaseInfo.curCount); // 剩余秒数
                my_printf("curColour               %d\n", car_self_phaseInfo.curColour); //  1-红灯 2-绿灯 3-黄灯
                my_printf("nextGreenStartTime      %d\n", car_self_phaseInfo.nextGreenStartTime); // 一直为0
                my_printf("nextGreenDuration       %d\n", car_self_phaseInfo.nextGreenDuration); // 一直为0
                my_printf("-----------------------------------------------\n")
                spat_print_time_cnt = 0;
            }
            else
            {
                spat_print_time_cnt++;
            }

        }
    }
    //printf("car_self_phaseInfo.phaseId %d \n", car_self_phaseInfo.phaseId);

    if (car_self_phaseInfo.phaseId)
    {
        pstoCarInfoList->stoCarInfos[pstoCarInfoList->num].u8type = SUGG_TYPE_signlight;
        pstoCarInfoList->stoCarInfos[pstoCarInfoList->num].u8phaseId = (uint8_t)pbsmSelfInfo->stposOnMap.phaseId_original;
        pstoCarInfoList->stoCarInfos[pstoCarInfoList->num].u8phaseColor = (uint8_t)car_self_phaseInfo.curColour;
        pstoCarInfoList->stoCarInfos[pstoCarInfoList->num].u8phaseTime = (uint8_t)car_self_phaseInfo.curCount;
        pstoCarInfoList->stoCarInfos[pstoCarInfoList->num].u16event_distance = (uint16_t)pbsmSelfInfo->stposOnMap.disFromCrossing;

        uint16_t sugg_speed = (uint16_t)pbsmSelfInfo->speed;
        //- 如果前方是绿灯
        if (2 == car_self_phaseInfo.curColour)
        {
            // 如果剩余秒数等于0则置为1防止后面除数为0
            car_self_phaseInfo.curCount = car_self_phaseInfo.curCount == 0 ? 1 : car_self_phaseInfo.curCount;
            //? 建议速度为到路口的距离除以当前剩余秒数，会不会太极限了？
            sugg_speed = pbsmSelfInfo->stposOnMap.disFromCrossing / car_self_phaseInfo.curCount;
            // 如果计算出来的建议速度小于当前速度，那没必要减速，原速行驶即可
            if (sugg_speed < pbsmSelfInfo->speed)
            {
                sugg_speed = pbsmSelfInfo->speed;
            }
            // 如果计算出来的建议速度大于车道限速，则将建议速度降低到60%
            if (sugg_speed > pbsmSelfInfo->stposOnMap.speed_max)
            {
                //sugg_speed = pbsmSelfInfo->stposOnMap.disFromCrossing / car_self_phaseInfo.nextGreenStartTime;
                sugg_speed = sugg_speed * 60 / 100;
            }
        }

        //- 如果前方是红灯或者黄灯
        if (1 == car_self_phaseInfo.curColour || 3 == car_self_phaseInfo.curColour)
        {
            // 建议速度为当前行驶速度降低到60%
            sugg_speed = pbsmSelfInfo->speed * 60 / 100;
        }

        pstoCarInfoList->stoCarInfos[pstoCarInfoList->num].u16sugg_speed = sugg_speed;
        pstoCarInfoList->num++; //. 后面是下一个toCarInfo了

        //- 如果前方是红灯或者黄灯 
        if (1 == car_self_phaseInfo.curColour || 3 == car_self_phaseInfo.curColour)
        {
            // 如果剩余秒数等于0则置为1防止后面除数为0
            pbsmSelfInfo->speed = pbsmSelfInfo->speed == 0 ? 1 : pbsmSelfInfo->speed;
            // 计算到达路口需要的时间
            int timeFromCrossing = pbsmSelfInfo->stposOnMap.disFromCrossing / pbsmSelfInfo->speed;
            // 时间阈值初始化为当前剩余秒数
            int timeThreshold = car_self_phaseInfo.curCount;
            //? 如果是黄灯的话时间阈值+3（为什么？）
            if (3 == car_self_phaseInfo.curColour)
            {
                timeThreshold = timeThreshold + 3;
            }

            //printf("timeFromCrossing %d, timeThreshold %d\n", timeFromCrossing, timeThreshold);
            //- 如果到达路口需要的时间小于时间阈值，发送闯红灯警告
            if (timeFromCrossing < timeThreshold)
            {
                printf("SUGG_TYPE_red_light\n");
                pstoCarInfoList->stoCarInfos[pstoCarInfoList->num].u8type = SUGG_TYPE_red_light;
                pstoCarInfoList->stoCarInfos[pstoCarInfoList->num].u16event_distance = (uint16_t)pbsmSelfInfo->stposOnMap.disFromCrossing;
                // 建议速度为当前速度降低到60%
                pstoCarInfoList->stoCarInfos[pstoCarInfoList->num].u16sugg_speed = sugg_speed = pbsmSelfInfo->speed * 60 / 100;
                pstoCarInfoList->num++;
            }
        }
        //- 如果前方是绿灯
        if (2 == car_self_phaseInfo.curColour)
        {
            pbsmSelfInfo->speed = pbsmSelfInfo->speed == 0 ? 1 : pbsmSelfInfo->speed;
            int timeFromCrossing = pbsmSelfInfo->stposOnMap.disFromCrossing / pbsmSelfInfo->speed;
            int timeThreshold = car_self_phaseInfo.curCount;

            //printf("timeFromCrossing %d, timeThreshold %d\n", timeFromCrossing, timeThreshold);
            //- 如果到达路口的时间大于时间阈值，发送闯红灯警告 
            if (timeFromCrossing > timeThreshold)
            {
                my_printf("SUGG_TYPE_red_light\n");
                pstoCarInfoList->stoCarInfos[pstoCarInfoList->num].u8type = SUGG_TYPE_red_light;
                pstoCarInfoList->stoCarInfos[pstoCarInfoList->num].u16event_distance = (uint16_t)pbsmSelfInfo->stposOnMap.disFromCrossing;
                //? 建议速度为当前速度降低到60%，可能是为了等下一个绿灯？
                pstoCarInfoList->stoCarInfos[pstoCarInfoList->num].u16sugg_speed = sugg_speed = pbsmSelfInfo->speed * 60 / 100;
                pstoCarInfoList->num++;
            }
        }
    }

    return 0;
}
```

## 任务处理函数

``` C++
void task1(qBsmInfo *pbsmSelfInfo, qSpatInfo *pspatInfo) 
{
  /* 获取车辆自身的相位信息 */
  qPhaseAttr car_self_phaseInfo;
  for(uint32_t i=0;i<pspatInfo->PhaseNum;i++) 
  {
    if (pbsmSelfInfo->stposOnMap.phaseId_original == pspatInfo->stPhaseInfo[i].phaseId)   
    {
      memcpy(&car_self_phaseInfo, &pspatInfo->stPhaseInfo[i], sizeof(qPhaseAttr));
    }
  }
  uint32_t sugg_speed = 0; // 建议速度
  if(car_self_phaseInfo.phaseId)
  {
    // 车辆当前速度，如果为0，则使其为1，防止除数为0
    uint32_t self_speed = pbsmSelfInfo->speed == 0 ? 1 : pbsmSelfInfo->speed;
    // 车辆到达交叉口的时间，四舍五入取整数
    uint32_t timeToCrossing = (uint32_t)((float)pbsmSelfInfo->stposOnMap.disFromCrossing / (float)self_speed + 0.5);
    printf("车辆保持当前速度到达交叉口的时间 = %ds\n", timeToCrossing);
    /* 如果当前相位为绿灯，根据信号灯剩余时间调整建议车速 */
    if(2 == car_self_phaseInfo.curColour)
    {
      /* 如果到达交叉口的时间小于绿灯剩余时间加黄灯时间，说明可以通过，保持当前速度 */ 
      if(timeToCrossing < car_self_phaseInfo.curCount + 3)
      {
        sugg_speed = pbsmSelfInfo->speed;
      }
      /* 如果到达交叉口的时间大于等于绿灯剩余时间加黄灯时间，说明无法通过，需要加速。如果加速到车道最高速度依然无法通过，则计算下一次绿灯的绿波速度 */
      else
      {
        sugg_speed = pbsmSelfInfo->stposOnMap.disFromCrossing / (car_self_phaseInfo.curCount + 3);
        /* 如果算出建议速度小于等于车道限速，则建议速度合理；
          如果大于车道限速，则考虑下一次绿灯中值的绿波速度 */
        if(sugg_speed > pbsmSelfInfo->stposOnMap.speed_max)
        {
          // 距离下一次绿灯的中间时刻还有多久
          uint32_t nextGreenMiddleTime = car_self_phaseInfo.nextGreenStartTime + car_self_phaseInfo.nextGreenDuration / 2;
          // 如果算出该时间为0，则将其置为1，防止除数为0
          nextGreenMiddleTime = nextGreenMiddleTime == 0 ? 1 : nextGreenMiddleTime;
          sugg_speed = pbsmSelfInfo->stposOnMap.disFromCrossing / nextGreenMiddleTime;
        }
      }
    }
    /* 如果当前相位为红灯或黄灯，则考虑下一次绿灯中值的绿波速度 */
    else if(1 == car_self_phaseInfo.curColour || 3 == car_self_phaseInfo.curColour)
    {
      // 距离下一次绿灯的中间时刻还有多久
      uint32_t nextGreenMiddleTime = car_self_phaseInfo.nextGreenStartTime + car_self_phaseInfo.nextGreenDuration / 2;
      nextGreenMiddleTime = nextGreenMiddleTime == 0 ? 1 : nextGreenMiddleTime;
      sugg_speed = pbsmSelfInfo->stposOnMap.disFromCrossing / nextGreenMiddleTime;
      /* 如果建议速度大于车道限速，则建议速度设为车道限速 */
      if(sugg_speed > pbsmSelfInfo->stposOnMap.speed_max)
      {
        sugg_speed = pbsmSelfInfo->stposOnMap.speed_max;
      }
    }
  }
  printf("车辆建议速度 = %ddm/s\n", sugg_speed);


  ToCarInfo_data stToCarInfo;
  stToCarInfo.startId = ntohs(0x55aa);
  stToCarInfo.mode = 1;
  stToCarInfo.dataLen = ntohs(10);
  stToCarInfo.type = SUGG_TYPE_signlight;
  stToCarInfo.phaseId = car_self_phaseInfo.phaseId;
  stToCarInfo.phaseColor = car_self_phaseInfo.curColour;
  stToCarInfo.phaseTime = car_self_phaseInfo.curCount;
  stToCarInfo.distance = ntohs(pbsmSelfInfo->stposOnMap.disFromCrossing);
  stToCarInfo.suggSpeed = ntohs(sugg_speed);
  stToCarInfo.u8is_merge = 0;
  stToCarInfo.u8intersection_turn = 0;
  stToCarInfo.endId = ntohs(0xaa55);
  
  sendByUartToCar((char*)&stToCarInfo, sizeof(ToCarInfo_data));
}
```
